1. 统计二叉树中平衡路径的数量

路径可从任意节点出发，但只能向下延伸；要求路径节点和为 0，且路径长度至少为 2。

核心思路
DFS + 前缀和计数
时间复杂度
O(n)
空间复杂度
O(h)

这题本质上是“向下路径和”变体。沿根到当前节点维护前缀和，当某个前缀和值重复出现时，就意味着中间那一段向下路径的和为 0。额外注意把长度为 1 的单节点路径剔除掉。

from collections import deque, defaultdict
import ast
import sys


class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right


def build_tree(level_list):
    if not level_list or level_list[0] is None:
        return None

    root = TreeNode(level_list[0])
    q = deque([root])
    i = 1

    while q and i < len(level_list):
        node = q.popleft()

        if i < len(level_list) and level_list[i] is not None:
            node.left = TreeNode(level_list[i])
            q.append(node.left)
        i += 1

        if i < len(level_list) and level_list[i] is not None:
            node.right = TreeNode(level_list[i])
            q.append(node.right)
        i += 1

    return root


def count_balanced_paths(root):
    # 统计“从任意节点出发、仅向下、节点和为 0、长度至少为 2”的路径数
    prefix_count = defaultdict(int)
    prefix_count[0] = 1  # 空前缀
    ans = 0

    def dfs(node, prefix_sum):
        nonlocal ans
        if not node:
            return

        cur_sum = prefix_sum + node.val

        # 所有以前出现过的相同前缀，都对应一条以当前节点结尾、和为 0 的向下路径
        ans += prefix_count[cur_sum]

        # 但如果当前节点值本身就是 0，会把“长度为 1 的单节点路径”也算进去，需去掉
        if node.val == 0:
            ans -= 1

        prefix_count[cur_sum] += 1
        dfs(node.left, cur_sum)
        dfs(node.right, cur_sum)
        prefix_count[cur_sum] -= 1

    dfs(root, 0)
    return ans


def parse_input(text):
    text = text.strip()
    if not text:
        return []

    # 支持两种输入：
    # 1) [1, -1, 0, None, 2]
    # 2) 1 -1 0 None 2
    try:
        data = ast.literal_eval(text)
        if isinstance(data, list):
            return data
    except Exception:
        pass

    vals = []
    for token in text.replace(',', ' ').split():
        if token.lower() == 'none':
            vals.append(None)
        else:
            vals.append(int(token))
    return vals


def func():
    data = sys.stdin.read()
    level_list = parse_input(data)
    root = build_tree(level_list)
    print(count_balanced_paths(root))


if __name__ == '__main__':
    func()

2. 网络异常流量传播链路溯源

先筛选关键节点，再根据空间关联和时间先后建立有向边，最后求传播链条可覆盖的最大用户数。

核心思路
关键节点筛选 + DAG 最长路
时间复杂度
O(n²)
空间复杂度
O(n²)

因为链路只会从更早时间指向更晚时间，所以图天然是 DAG。先按曼哈顿距离构建邻接关系并计算邻域负载和，筛出关键节点；再按时间排序做动态规划，求至少包含 2 个节点的最大 users 总和。

import sys


def func():
    data = sys.stdin.read().strip().split()
    if not data:
        print(0)
        return

    it = iter(data)
    n = int(next(it))
    epsilon = int(next(it))
    w_threshold = int(next(it))

    nodes = []
    for _ in range(n):
        x = int(next(it))
        y = int(next(it))
        t = int(next(it))
        w = int(next(it))
        users = int(next(it))
        nodes.append((x, y, t, w, users))

    # 1) 先判断关键节点：邻域内（含自身）负载和 >= 阈值
    neighbor_load_sum = [0] * n
    adjacent = [[False] * n for _ in range(n)]

    for i in range(n):
        xi, yi, _, _, _ = nodes[i]
        for j in range(i, n):
            xj, yj, _, _, _ = nodes[j]
            if abs(xi - xj) + abs(yi - yj) <= epsilon:
                adjacent[i][j] = adjacent[j][i] = True
                neighbor_load_sum[i] += nodes[j][3]
                if i != j:
                    neighbor_load_sum[j] += nodes[i][3]

    key_nodes = [i for i in range(n) if neighbor_load_sum[i] >= w_threshold]
    if len(key_nodes) < 2:
        print(0)
        return

    # 2) 在关键节点之间建有向图：邻接且时间更早 -> 时间更晚
    key_nodes.sort(key=lambda idx: nodes[idx][2])  # 按时间升序
    k = len(key_nodes)

    dp = [0] * k  # 以第 i 个关键节点结尾的最大 users 和
    has_in_edge = [False] * k
    has_any_edge = False

    for i in range(k):
        idx_i = key_nodes[i]
        dp[i] = nodes[idx_i][4]  # 至少可以从自己开始，后面会专门判断是否真的形成链路
        ti = nodes[idx_i][2]

        for j in range(i):
            idx_j = key_nodes[j]
            tj = nodes[idx_j][2]
            if tj < ti and adjacent[idx_j][idx_i]:
                has_in_edge[i] = True
                has_any_edge = True
                dp[i] = max(dp[i], dp[j] + nodes[idx_i][4])

    if not has_any_edge:
        print(0)
        return

    # 只统计真正由至少 2 个节点构成的链路
    ans = 0
    for i in range(k):
        if has_in_edge[i]:
            ans = max(ans, dp[i])

    print(ans)


if __name__ == '__main__':
    func()
